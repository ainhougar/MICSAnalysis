#Set-up: Turn variables into factor to keep labels in tbl_summary 
factoritems<-function(data_f, varsf){ 
  
  data_f<-data_f%>%mutate(across(all_of(varsf), to_factor))#Important to consider labels across functions. 
  
  data_f<-as.data.frame(data_f)
  return(data_f)
  
}

numericitems<-function(data_f, varsf){ 
  
  data_f<-data_f%>%mutate(across(all_of(varsf), as.numeric))#Important to consider labels across functions. 
  
  data_f<-as.data.frame(data_f)
  return(data_f)
  
}

#Description table function:::::::::::::::::
desc_table<-function(data_f,vars_f){ 
  data_sum<-data_f%>%
    tbl_summary(
      include =   all_of(vars_f), 
      type = all_categorical() ~ "categorical",   # Show all levels for binary variables 
      statistic = list(all_continuous() ~ "{mean} ({sd})", 
                       all_categorical()~ "{n} ({p}%)") ,
      digits = all_continuous() ~ 2)%>%
    bold_labels()%>%
    #Add n per variable
    add_n(
      statistic = "{N_nonmiss}",
      col_label = "**N**",
      footnote = FALSE)
  return(data_sum)
}
