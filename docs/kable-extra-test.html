Balo<-readRDS("DATA/Created_Datasets/mBalo.rds")
Pahktu<-readRDS("DATA/Created_Datasets/mPahktu.rds")

#Create filtered datasets
Balo_f<-Balo%>%filter(!is.na(UF17))
Pahktu_f<-Pahktu%>%filter(!is.na(UF17))

ECDIvars<-c("EC8", "EC9","EC10",
            "EC11", "EC12", "EC13",
            "EC14", "EC15")

Balo_f<-Balo_f%>%mutate(across(all_of(ECDIvars), to_factor))#Important to consider labels across functions. 
Pahktu_f<-Pahktu_f%>%mutate(across(all_of(ECDIvars), to_factor))
Balo_sum<-Balo_f%>%
  tbl_summary(
    include =   all_of(ECDIvars), 
    statistic = list(all_continuous() ~ "{mean} ({sd})", 
                     all_categorical()~ "{n} ({p}%)") ,
    digits = all_continuous() ~ 2)%>%
  bold_labels()
