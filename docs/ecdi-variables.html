#Goal: Creation of ECDI 
#The ECDI consists of 4 dimensions: (1)Literacy-numeracy, (2)Physical, (3)Social-emotional, (4)Approaches to learning 
#ECDI items are coded such that answers to questions are 1: YES & 2: NO
#The ECDI is a binary variable: (a)Being developmentally on track VS (b)Not being on track 


#(1)Literacy-numeracy on track: 
#Vars: EC6, EC7, EC8
#Developmentally on track: Having a score of 1 in at least 2 out of the 3 items. 

#(2)Physical
#Developmentally on track: EC9=1 OR EC10=2

#(3)Approaches to learning:  EC9=1 OR EC10=1

#(4)Social-emotional: At least 2 out of the three received these scores: EC13=1 , EC14=2 , EC15=2

#CREATING VARIABLES DEVELOPMENTALLY ON TRACK VS NOT 

#Filter by individuals who have information across all variables


filtercreateECDI<-function(data){ 

  data <- data %>%
    filter(if_all(all_of(ECDIvars), ~ !is.na(.))) #if_all in filter. across in mutate()
  
  data <- data %>%
    mutate(
      # Domain 1: Literacy-numeracy — at least 2 out of 3 are 1 (YES)
      literacy_track = rowSums(across(c(EC6, EC7, EC8), ~ . == 1), na.rm = TRUE) >= 2,
      
      # Domain 2: Physical — EC9 == 1 OR EC10 == 2
      physical_track = case_when(EC9 == 1 | EC10 == 2 ~"TRUE", 
                                 TRUE~"FALSE"),
      
      # Domain 3: Approaches to learning — EC11 == 1 OR EC12 == 1
      learning_track = case_when(EC11 == 1 | EC12 == 1 |EC11 == 1 & EC12 == 1  ~ "TRUE",
                                 TRUE~"FALSE"),
      
      # Domain 4: Social-emotional — at least 2 out of 3 specific conditions
      socioemotional_track = rowSums(cbind(EC13 == 1, EC14 == 2, EC15 == 2), na.rm = TRUE) >= 2,
      
      # Final ECDI: Developmentally on track if ALL 4 domains are TRUE
      ECDI = rowSums(cbind(socioemotional_track == TRUE, learning_track  == "TRUE", 
                           physical_track == "TRUE", literacy_track == TRUE),
                          , na.rm = TRUE) >= 3
    ) 

return(data)
}
